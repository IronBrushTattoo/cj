      button "Submit" do
        #alert "Making sheets from #{@file} from last #{@days.text} days"

        #set_variables(@days.text)

        if confirm("Make sheets from #{@file} from last #{@days} days?")
          set_variables(@days)
          Sheets.make_sheets(@file)
        end
      end

      button "X" do
        exit()
